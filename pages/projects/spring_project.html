Spring Boot CRM System

An enterprise Customer Relationship Management system with role-based authentication, full CRUD, JPA/Hibernate ORM, and a PostgreSQL database.

Features

  • Customer management with full CRUD
  • Role-based authentication (ADMIN / USER)
  • Spring Security with JWT
  • JPA repositories with Hibernate ORM
  • Bean Validation on all inputs
  • PostgreSQL relational database
  • RESTful API with proper HTTP status codes

REST Controller

@RestController
@RequestMapping("/api/v1/customers")
@RequiredArgsConstructor
public class CustomerController {

    private final CustomerService service;

    @GetMapping
    public ResponseEntity<List<Customer>> getAll() {
        return ResponseEntity.ok(service.findAll());
    }

    @GetMapping("/{id}")
    public ResponseEntity<Customer> getById(@PathVariable Long id) {
        return ResponseEntity.ok(service.findById(id));
    }

    @PostMapping
    @PreAuthorize("hasRole('ADMIN')")
    public ResponseEntity<Customer> create(
            @Valid @RequestBody Customer customer) {
        return ResponseEntity.status(201).body(service.save(customer));
    }

    @DeleteMapping("/{id}")
    @PreAuthorize("hasRole('ADMIN')")
    public ResponseEntity<Void> delete(@PathVariable Long id) {
        service.delete(id);
        return ResponseEntity.noContent().build();
    }
}

JPA Entity

@Entity
@Table(name = "customers")
public class Customer {

    @Id @GeneratedValue(strategy = GenerationType.IDENTITY)
    private Long id;

    @NotBlank private String name;
    @Email    private String email;
    @NotBlank private String phone;

    @Enumerated(EnumType.STRING)
    private CustomerStatus status;
}
